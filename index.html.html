        /**

         * Calculates the total price and applies discounts based on the quantity of items.

        function calculateTotal() {

            let subtotal = 0;

            let totalItems = 0;

            // Calculate subtotal and total items

            for (const itemName in itemQuantities) {

                const quantity = itemQuantities[itemName];

                if (quantity > 0) {

                    const item = menuItems.find(item => item.name === itemName);

                    if (item) {

                        subtotal += item.price * quantity;

                        totalItems += quantity;

                    }

                }

            }

            // Sort policy by minItems in descending order to ensure correct application for ranges

            for (const policy of sortedPolicy) {

                if (totalItems >= policy.minItems && totalItems <= policy.maxItems) {

                    discountPercentage = policy.discount;

                    break; // Found the highest applicable discount

                }

            }

            // Calculate discount amount

            discountAmount = subtotal * discountPercentage;

                chatHistory.push({ role: "user", parts: [{ text: prompt }] });

                const payload = { contents: chatHistory };

                const apiKey = ""; // Canvas will provide this API key

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                const result = await response.json();

                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const text = result.candidates[0].content.parts[0].text;

                    modalContent.innerHTML = `<p>${text}</p>`;

                } else {

                    modalContent.innerHTML = `<p class="text-red-500">Could not retrieve health benefits. Please try again.</p>`;

                }

                modalContent.innerHTML = `<p class="text-red-500">An error occurred: ${error.message}. Please try again.</p>`;

            }

        }

        /**

         * Fetches ingredient substitution suggestions from the Gemini API.

         * @param {string} itemName - The name of the menu item.

         * @param {string} originalIngredients - The original ingredients of the menu item.

         * @param {string} ingredientToSubstitute - The ingredient the user wants to substitute.

         */

        async function fetchSubstitutionSuggestion(itemName, originalIngredients, ingredientToSubstitute) {

            substitutionResult.innerHTML = `

                <div class="loading-spinner"></div>

                <p class="text-center text-gray-500 mt-2">Generating suggestion...</p>

            `;

            const prompt = `For the drink "${itemName}" which contains these ingredients: ${originalIngredients}. If a customer wants to substitute "${ingredientToSubstitute}", what would be a healthy and suitable alternative? Provide 1-2 suggestions and a very brief reason for each. Keep it concise.`;

            const chatHistory = [];

            chatHistory.push({ role: "user", parts: [{ text: prompt }] });

            const payload = { contents: chatHistory };

            const apiKey = ""; // Canvas will provide this API key

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

            try {

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                const result = await response.json();

                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const text = result.candidates[0].content.parts[0].text;

                    substitutionResult.innerHTML = `<p>${text}</p>`;

                } else {

                    substitutionResult.innerHTML = `<p class="text-red-500">Could not retrieve substitution suggestion. Please try again.</p>`;

                }

            } catch (error) {

                console.error("Error fetching substitution suggestion:", error);

                substitutionResult.innerHTML = `<p class="text-red-500">An error occurred: ${error.message}. Please try again.</p>`;

            }

        }
